public class SecurityHeadersFilter : ActionFilterAttribute { public override void OnResultExecuting(HttpActionContext actionContext) { var headers = actionContext.Response.Headers; headers.Add("X-Frame-Options", "SAMEORIGIN"); headers.Add("X-Content-Type-Options", "nosniff"); headers.Add("X-XSS-Protection", "1; mode=block"); headers.Add("Content-Security-Policy", "default-src 'self'; frame-ancestors 'self';"); base.OnResultExecuting(actionContext); } } config.Filters.Add(new SecurityHeadersFilter()); [RoutePrefix("api/files")] public class FilesController : ApiController { private readonly TuDbContext _context = new TuDbContext(); [HttpGet, Route("tree")] public IHttpActionResult GetTree() { var archivos = _context.Archivos.ToList(); var nodos = archivos.Select(a => new { id = a.Id.ToString(), parent = a.PadreId.HasValue ? a.PadreId.ToString() : "#", text = a.Nombre }); return Ok(nodos); } [HttpGet, Route("pdf/{id}")] public HttpResponseMessage GetPdf(int id) { var archivo = _context.Archivos.Find(id); if (archivo == null || !File.Exists(archivo.RutaFisica)) return Request.CreateResponse(HttpStatusCode.NotFound); var stream = new FileStream(archivo.RutaFisica, FileMode.Open, FileAccess.Read); var response = new HttpResponseMessage(HttpStatusCode.OK) { Content = new StreamContent(stream) }; response.Content.Headers.ContentType = new MediaTypeHeaderValue("application/pdf"); response.Content.Headers.ContentDisposition = new ContentDispositionHeaderValue("inline") { FileName = Path.GetFileName(archivo.RutaFisica) }; return response; } }
